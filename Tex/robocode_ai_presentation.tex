\begin{frame}{State Definition (16 Features)}
\begin{itemize}
  \item Self: normalized energy, speed, heading as $\sin/\cos$
  \item Position: normalized $x,y$ and distances to four walls
  \item Enemy: bearing as $\sin/\cos$, normalized distance, enemy energy
  \item Temporal: $\Delta$bearing and $\Delta$distance
  \item Weapon readiness: normalized gun heat
  \item Freshness: 1 if enemy scanned recently, else 0
\end{itemize}

\[
s_t \in \mathbb{R}^{16}
\]
\end{frame}

%----------------5-----------------
\begin{frame}{How DQN Works (Value-Based)}
\begin{itemize}
  \item Network predicts $Q_\theta(s,a)$ for all actions.
  \item Policy uses epsilon-greedy exploration.
  \item Exploit step picks action with maximum predicted value.
\end{itemize}

\[
\pi(s_t)=\arg\max_a Q_\theta(s_t,a)
\]

\[
\epsilon_t=\epsilon_{\min}+\left(\epsilon_{\max}-\epsilon_{\min}\right)\exp\left(-\frac{\text{steps}}{\text{decay}}\right)
\]

\begin{itemize}
  \item Network: MLP $16 \rightarrow 128 \rightarrow 128 \rightarrow 7$.
  \item New evaluation mode: online learning disabled and fixed $\epsilon$ (typically $0.0$).
\end{itemize}
\end{frame}

%----------------6-----------------
\begin{frame}{Target, Loss, and Soft Update}
\begin{itemize}
  \item Mini-batches sampled from replay memory.
  \item Terminal transitions have zero next-state bootstrap value.
\end{itemize}

\[
y_t = r_t + \gamma (1-d_t) \max_{a'} Q_{\bar\theta}(s_{t+1},a')
\]

\[
\mathcal{L}(\theta)=\mathbb{E}\left[\operatorname{SmoothL1}\left(Q_\theta(s_t,a_t),y_t\right)\right]
\]

\[
\bar\theta \leftarrow \tau\,\theta + (1-\tau)\,\bar\theta
\]

\begin{itemize}
  \item We use Adam, gradient clipping, and $\tau=0.005$.
\end{itemize}
\end{frame}

%----------------8-----------------
\begin{frame}{PPO Bot Overview (Implemented)}
\begin{itemize}
  \item Uses same state and action dimensions as DQN: $16$ features, $7$ actions.
  \item Network is actor-critic with shared trunk and two heads:
  \begin{itemize}
    \item policy head: action logits
    \item value head: $V(s)$ baseline
  \end{itemize}
  \item Trains from on-policy trajectories collected during episodes.
\end{itemize}

\[
\pi_\theta(a\mid s)=\operatorname{softmax}(\text{logits}_\theta(s)),\qquad
V_\phi(s)=\text{value head}(s)
\]
\end{frame}

%----------------9-----------------
\begin{frame}{PPO Objective and Update Rule}
\begin{itemize}
  \item Advantages are computed from returns minus value estimates.
  \item PPO clips policy ratio updates to limit destructive policy shifts.
\end{itemize}

\[
A_t = R_t - V(s_t),\qquad
r_t(\theta)=\frac{\pi_\theta(a_t\mid s_t)}{\pi_{\theta_{\text{old}}}(a_t\mid s_t)}
\]

\[
\mathcal{L}^{\text{clip}}(\theta)=\mathbb{E}\left[\min\left(r_t(\theta)A_t,
\operatorname{clip}(r_t(\theta),1-\epsilon,1+\epsilon)A_t\right)\right]
\]

\begin{itemize}
  \item Value loss is optimized jointly with policy objective.
  \item Current implementation runs multiple update epochs per episode batch.
\end{itemize}
\end{frame}

%----------------10-----------------
\begin{frame}{DQN vs PPO in This Project}
\begin{columns}[T,totalwidth=\textwidth]
\column{0.5\textwidth}
\textbf{DQN (off-policy)}
\begin{itemize}
  \item Replay memory
  \item Target network + soft update
  \item Q-value regression (SmoothL1)
  \item Epsilon-greedy exploration
\end{itemize}

\column{0.5\textwidth}
\textbf{PPO (on-policy)}
\begin{itemize}
  \item Trajectory batches
  \item Actor-critic objective
  \item Clipped policy updates
  \item Stochastic sampling from policy
\end{itemize}
\end{columns}

\vspace{0.2cm}
\small This gives two RL baselines with shared state/action design and different optimization behavior.
\end{frame}

%----------------12-----------------
\begin{frame}{Interpretation and ML-Safe Next Changes}
\begin{itemize}
  \item The policy generalizes strongly to Target, but struggles against movement-heavy opponents (SpinBot, Walls).
  \item Big train/eval drop on SpinBot suggests overfitting to current curriculum distribution.
  \item Low Walls performance indicates poor wall-management behavior under pressure.
\end{itemize}
\end{frame}

\begin{frame}{Possible changes}
\begin{itemize}
  \item Rebalance curriculum: increase sampling weight for Walls and SpinBot in training passes.
  \item Reward shaping: add stronger penalty for wall collisions and getting pinned near boundaries.
  \item Add action options for wall escape (turn-away + burst movement) while keeping RL policy control.
  \item Tune exploration schedule and evaluate with fixed $\epsilon=0$ on held-out opponents only.
\end{itemize}
\end{frame}
